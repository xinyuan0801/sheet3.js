Remove and adding row

user can remove row by index or add row by index(only available for horzontial headers table)

  • Java
  • Html 
      
        let addData = {
          first_name: "Marijo",
          last_name: "Durning",
          email: "mdurning0@loc.gov",
        };
        var t8 = new sheet3({
          id: "t8",
          row: 5,
          col: 3,
          type: "edit",
          location: "rowOpDemo",
          style: "MD",
          mode: "horzontial",
          data: JsontoRead
        });
        t8.createTable();
        
        function doDelete() {
          let row = $("#drow").val()
          t8.removeRow(parseInt(row))
        }
        
        function doAdd() {
          let row = $("#arow").val()
          t8.addRow(addData, row)
        }
      
    
        
          <div id="rowOpDemo"></div>
          <input type="text" id="drow" placeholder="row num(start 1)">
          <button type="button" class="btn btn-primary" id="Button" onclick="doDelete()">delete row
          <input type="text" id="arow" placeholder="row num(start 1)">
          <button type="button" class="btn btn-primary" id="Button" onclick="doAdd()">add row

Search row

user can search specific row with a specific column and the content to search

  • Java
  • Html 
      
        const NumberTest = [
          { first_name: "Leonard", last_name: "Faulconer", age: 45 },
          { first_name: "Editha", last_name: "Keogh", age: 49 },
          { first_name: "Jeth", last_name: "Hasson", age: 83 },
          { first_name: "Liz", last_name: "De Bruyn", age: 46 },
          { first_name: "Valentia", last_name: "Lasslett", age: 40 },
        ];
        
        var t10 = new sheet3({
          id: "t10",
          row: 5,
          col: 3,
          type: "edit",
          location: "searchDemo",
          style: "MD",
          mode: "horzontial",
          data: NumberTest
        });
        t10.createTable();
        
        function doFind() {
          let selectedCol = $('#searchCol').val();
          let searchContent = $('#searchContent').val();
          $(t10.search(selectedCol, searchContent)).css('background-color', '#ddd')
        }
        
        function doReset() {
          let table = document.getElementById('t10')
          $(table).remove();
          t10.createTable();
        }
      
    
        
          <div id="searchDemo" class="demoTable"></div>
          <div id="inputGroups">
          <input type="text" id="searchCol" placeholder="header's name">
          </div>
          <div id="inputGroups">
          <input type="text" id="searchContent" placeholder="content">
          <button type="button" class="btn btn-primary" onclick="doFind()">Find
          <button type="button" class="btn btn-primary" onclick="doReset()">reset
          </div>

Show specific rows

user can choose to only show the rows that is in the range of row number given by user

  • Java
  • Html 
      
        const NumberTest = [
          { first_name: "Leonard", last_name: "Faulconer", age: 45 },
          { first_name: "Editha", last_name: "Keogh", age: 49 },
          { first_name: "Jeth", last_name: "Hasson", age: 83 },
          { first_name: "Liz", last_name: "De Bruyn", age: 46 },
          { first_name: "Valentia", last_name: "Lasslett", age: 40 },
        ];
        
        var t12 = new sheet3({
          id: "t12",
          row: 5,
          col: 3,
          type: "edit",
          location: "showRowDemo",
          style: "MD",
          mode: "horzontial",
          data: NumberTest
        });
        t12.createTable();
        
        function doShow() {
          t12.resetShow();
          let top = $('#toprow').val();
          let bottom = $('#botrow').val();
          t12.showRow(parseInt(top), parseInt(bottom))
        }
        
        function doShowReset() {
          t12.resetShow();
        }
      
    
        

          <div id="showRowDemo" class="demoTable">
<div id="inputGroups"> <input type="text" id="toprow" placeholder="row num(start 1)"> </div> <div id="inputGroups"> <input type="text" id="botrow" placeholder="row num(start 1)"> <button type="button" class="btn btn-primary" onclick="doShow()">show <button type="button" class="btn btn-primary" onclick="doShowReset()">reset
